--  PANDUAN INSTALASI & MENJALANKAN PROGRAM
1. Instalasi Library (Melalui Terminal/CMD)
    Sebelum menjalankan program, pastikan Anda telah menginstal library Python yang diperlukan. Buka CMD atau Terminal, lalu jalankan perintah berikut secara berurutan:

    Bash nama File: 
    pip install flask
    pip install mysql-connector-python
    pip install Werkzeug
    (Atau Anda bisa menginstal semuanya sekaligus dengan perintah: pip install Flask Werkzeug mysql-connector-python)

2. Pengaturan Database (MySQL via XAMPP)
    Untuk menyiapkan database, ikuti langkah-langkah di bawah ini:

    Aktifkan Apache dan MySQL pada panel kontrol XAMPP.

    Buka browser dan akses dashboard database di: http://localhost/phpmyadmin/.

    Buat database baru (contoh nama: Larike_warehouse).

    Klik tab SQL pada bagian atas menu phpMyAdmin.

    Salin (Copy) kode SQL database yang ada di bawah ini (dari bagian START hingga END), lalu tempelkan (Paste) ke dalam kolom query SQL tersebut.

    Klik tombol Go atau Kirim untuk membuat tabel.

3. Cara Menjalankan Program
    Setelah library terinstal dan database siap:

    Buka folder proyek Anda, lalu cari file utama bernama app.py.

    Jalankan file tersebut dengan cara klik Run di editor (seperti VS Code/PyCharm) atau melalui CMD dengan perintah: python app.py.

    Jika muncul pesan error di awal, silakan coba klik Run kembali hingga muncul keterangan:

* Running on http://127.0.0.1:5000

    Salin alamat IP tersebut ke browser Anda untuk membuka website.

    Tekan CTRL+C pada keyboard jika ingin menghentikan program.

    Catatan: Pastikan koneksi database di file db_config.py sudah sesuai dengan username dan password MySQL Anda.

-- 1. Membuat Database dengan nama sesuai permintaan
CREATE DATABASE Larike_warehouse;
USE Larike_warehouse;

-- ==========================================
-- KELOMPOK TABEL MASTER (DATA INDUK)
-- ==========================================

-- Tabel user: Untuk pengelola sistem
CREATE TABLE user (
    id_user INT AUTO_INCREMENT PRIMARY KEY,
    username VARCHAR(50) NOT NULL,
    password VARCHAR(255) NOT NULL,
    email_user VARCHAR(100),
    no_hp_user VARCHAR(20),
    foto_user VARCHAR(255) DEFAULT 'default.png'
    role ENUM('Admin', 'Staff Gudang') NOT NULL
);

-- Tabel kategori: Untuk pengelompokan jenis barang
CREATE TABLE kategori (
    id_kategori INT AUTO_INCREMENT PRIMARY KEY,
    nama_kategori VARCHAR(100) NOT NULL
);

-- Tabel lokasi_rak: Lokasi fisik penyimpanan di gudang
CREATE TABLE lokasi_rak (
    id_lokasi INT AUTO_INCREMENT PRIMARY KEY,
    kode_rak VARCHAR(20) NOT NULL, -- Contoh: A-01
    area_gudang VARCHAR(100)       -- Contoh: Zona Pendingin
);

-- Tabel supplier: Pihak penyedia barang
CREATE TABLE supplier (
    id_supplier INT AUTO_INCREMENT PRIMARY KEY,
    nama_supplier VARCHAR(100) NOT NULL,
    email_supplier VARCHAR(100),
    no_hp_supplier VARCHAR(20),
    alamat_supplier TEXT
);

-- Tabel customer: Pihak tujuan barang keluar
CREATE TABLE customer (
    id_customer INT AUTO_INCREMENT PRIMARY KEY,
    nama_customer VARCHAR(100) NOT NULL,
    email_customer VARCHAR(100),
    no_hp_customer VARCHAR(20),
    alamat_customer TEXT
);

-- Tabel barang: Inti data stok
CREATE TABLE barang (
    id_barang INT AUTO_INCREMENT PRIMARY KEY,
    id_kategori INT,
    id_lokasi INT,
    nama_barang VARCHAR(200) NOT NULL,
    sku_code VARCHAR(50) UNIQUE,
    foto_barang VARCHAR(255), -- Menyimpan nama file gambar
    stok_saat_ini INT DEFAULT 0,
    harga_satuan DECIMAL(15, 2),
    satuan VARCHAR(20),      -- Contoh: Pcs, Box, Kg
    FOREIGN KEY (id_kategori) REFERENCES kategori(id_kategori) ON DELETE SET NULL,
    FOREIGN KEY (id_lokasi) REFERENCES lokasi_rak(id_lokasi) ON DELETE SET NULL
);

-- ==========================================
-- KELOMPOK TABEL TRANSAKSI (PERGERAKAN BARANG)
-- ==========================================

-- Tabel barang_masuk: Hubungan Barang + Supplier + User
CREATE TABLE barang_masuk (
    id_masuk INT AUTO_INCREMENT PRIMARY KEY,
    id_barang INT,
    id_supplier INT,
    id_user INT,
    jumlah_masuk INT NOT NULL,
    tanggal_masuk DATETIME DEFAULT CURRENT_TIMESTAMP,
    nomor_po VARCHAR(50),
    FOREIGN KEY (id_barang) REFERENCES barang(id_barang),
    FOREIGN KEY (id_supplier) REFERENCES supplier(id_supplier),
    FOREIGN KEY (id_user) REFERENCES user(id_user)
);

-- Tabel barang_keluar: Hubungan Barang + Customer + User
CREATE TABLE barang_keluar (
    id_keluar INT AUTO_INCREMENT PRIMARY KEY,
    id_barang INT,
    id_customer INT,
    id_user INT,
    jumlah_keluar INT NOT NULL,
    tanggal_keluar DATETIME DEFAULT CURRENT_TIMESTAMP,
    nomor_do VARCHAR(50),
    FOREIGN KEY (id_barang) REFERENCES barang(id_barang),
    FOREIGN KEY (id_customer) REFERENCES customer(id_customer),
    FOREIGN KEY (id_user) REFERENCES user(id_user)
);

-- Tabel stok_opname: Untuk audit kecocokan stok fisik dan sistem
CREATE TABLE stok_opname (
    id_opname INT AUTO_INCREMENT PRIMARY KEY,
    id_barang INT,
    id_user INT,
    stok_fisik INT NOT NULL,
    selisih INT NOT NULL,
    tanggal_audit DATETIME DEFAULT CURRENT_TIMESTAMP,
    keterangan TEXT,
    FOREIGN KEY (id_barang) REFERENCES barang(id_barang),
    FOREIGN KEY (id_user) REFERENCES user(id_user)
);
